Integration Example

Here's how to integrate SchedulifyX in your backend (Node.js example):

// Your SaaS Backend - Node.js Example
const SCHEDULIFY_API_KEY = process.env.SCHEDULIFY_API_KEY;
// Note: api.schedulifyx.com already includes /api/v1 prefix via nginx
const SCHEDULIFY_API_URL = 'https://api.schedulifyx.com';

// When your user wants to connect their social accounts
app.post('/api/connect-social/:platform', async (req, res) => {
  const userId = req.user.id; // Your authenticated user
  
  // 1. Check if user already has a tenant, if not create one
  let tenantId = await db.getTenantId(userId);
  
  if (!tenantId) {
    // Create tenant for this user
    const tenant = await fetch(`${SCHEDULIFY_API_URL}/tenants`, {
      method: 'POST',
      headers: {
        'Authorization': `Bearer ${SCHEDULIFY_API_KEY}`,
        'Content-Type': 'application/json',
      },
      body: JSON.stringify({
        externalId: userId,
        metadata: { email: req.user.email, plan: req.user.plan }
      }),
    }).then(r => r.json());
    
    tenantId = tenant.data.id;  // Response wrapped in { data: {...} }
    await db.saveTenantId(userId, tenantId); // Save mapping
  }
  
  // 2. Get OAuth URL for the platform
  const connectResult = await fetch(
    `${SCHEDULIFY_API_URL}/tenants/${tenantId}/connect/${req.params.platform}?redirectUri=${encodeURIComponent('https://yourapp.com/oauth/callback')}`,
    {
      headers: {
        'Authorization': `Bearer ${SCHEDULIFY_API_KEY}`,
      },
    }
  ).then(r => r.json());
  
  // 3. Redirect user to connect their account
  res.json({ authUrl: connectResult.data.url });
});

// When your user creates a post
app.post('/api/posts', async (req, res) => {
  const tenantId = await db.getTenantId(req.user.id);
  
  // mode: 'publish' (immediate), 'schedule' (timed), 'draft' (save only)
  const post = await fetch(`${SCHEDULIFY_API_URL}/posts`, {
    method: 'POST',
    headers: {
      'Authorization': `Bearer ${SCHEDULIFY_API_KEY}`,
      'Content-Type': 'application/json',
    },
    body: JSON.stringify({
      content: req.body.content,
      platforms: req.body.accountIds.map(id => ({
        accountId: id,
        platform: req.body.platform,
      })),
      tenantUserId: tenantId,  // Include tenant ID in the body
      mode: req.body.scheduledFor ? 'schedule' : 'publish',
      scheduledFor: req.body.scheduledFor || undefined,
    }),
  }).then(r => r.json());
  
  res.json(post.data);  // Response wrapped in { data: {...} }
});
